Unzip or checkout a version of CADuntu into a directory.
CD into that directory and follow these instructions:

# Compiling dxflib
cd dxflib
./configure
(Don't run make depend on OSX, this might fail)
make

cd ..

# Compile fparser
cd fparser
./configure
(Don't run make depend on OSX, this might fail)
make

cd ..

# Build makefile and compule CADuntu

qmake
make
